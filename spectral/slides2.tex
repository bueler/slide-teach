\begin{frame}{what happened in part I}

\begin{itemize}
\item see part I first: \quad \href{http://bueler.github.io/M617S20/slides1.pdf}{\texttt{bueler.github.io/M617S20/slides1.pdf}}
\item we proved the Schur decomposition, for all square matrices, and the spectral theorem for normal matrices:
    \begin{itemize}
    \item[] $A = Q T Q^*$ \quad \emph{Schur decomposition}
    \item[] $A = Q \Lambda Q^*$ \quad \emph{spectral theorem} when $AA^*=A^*A$
    \end{itemize}
where $Q$ is unitary ($Q^*Q=I$), $T$ is upper-triangular, and $\Lambda$ is diagonal
    \begin{itemize}
    \item[$\circ$] this spectral theorem for matrices is sometimes called the \emph{principal axis decomposition} (for quadratic forms)
    \end{itemize}
\item \emph{lemma.} a linear map $U$ on the inner product space $\CC^n$, where $\ip{v}{w}=v^*w$, is a unitary matrix (in the standard basis) if and only if it is linear, bijective, and an isometry
\item in infinite dimensions we will want to use unitary maps when possible, because such maps preserve both vector space and metric structure
    \begin{itemize}
    \item[$\circ$] with effort, we will be able to extend the spectral theorem
    \item[$\circ$] \dots but not the Schur decomposition
    \end{itemize}
\end{itemize}
\end{frame}
